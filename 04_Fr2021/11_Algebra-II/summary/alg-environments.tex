%   == Counters ==

\newenvironment{dfn}[1][]{%
\mdfsetup{
    frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=nordyellow]{\strut Definition #1};
    },
    linecolor=nordyellow,
}
\begin{mdframed}[]\relax}
{\end{mdframed}}
