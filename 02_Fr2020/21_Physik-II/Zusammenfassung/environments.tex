\newenvironment{definition}[1]{%
\mdfsetup{
    frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=ocean!35]{\strut Definition~#1};
    },
    innertopmargin=0pt,
    linecolor=ocean!35,
    linewidth=2pt,
    topline=true,
    frametitleaboveskip=\dimexpr-\ht\strutbox\relax
}
\begin{mdframed}[]\relax}
{\end{mdframed}}
