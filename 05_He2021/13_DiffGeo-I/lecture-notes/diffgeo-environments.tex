\mdfsetup{
  innertopmargin=0pt,
  linewidth=2pt,
  topline=true,
  frametitleaboveskip=\dimexpr-\ht\strutbox\relax
}

\newenvironment{dfn}[1][]{%
\mdfsetup{
    frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=orange!50]{\strut Definition #1};
    },
    linecolor=orange!50,
}
\begin{mdframed}[]\relax}
{\end{mdframed}}
